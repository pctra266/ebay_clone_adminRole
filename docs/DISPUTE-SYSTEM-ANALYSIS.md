public class DisputePriorityCalculator
{
    public string CalculatePriority(Dispute dispute, User buyer, User seller)
    {
        int score = 0;
        
        // 1. AMOUNT WEIGHT (30 points max)
        if (dispute.Amount >= 1000) score += 30;      // High value
        else if (dispute.Amount >= 500) score += 20;   // Medium value
        else if (dispute.Amount >= 100) score += 10;   // Low value
        else score += 5;                               // Very low
        
        // 2. TIME URGENCY (30 points max)
        var hoursLeft = (dispute.Deadline - DateTime.UtcNow).TotalHours;
        if (hoursLeft <= 24) score += 30;              // < 1 day
        else if (hoursLeft <= 72) score += 20;         // < 3 days
        else if (hoursLeft <= 168) score += 10;        // < 1 week
        
        // 3. BUYER PROFILE (20 points max)
        if (buyer.TotalSpent > 10000) score += 20;     // VIP buyer
        else if (buyer.TotalSpent > 5000) score += 15; // Good buyer
        else if (buyer.TotalSpent > 1000) score += 10; // Regular
        
        // 4. SELLER PROFILE (20 points max)
        if (seller.ViolationCount >= 5) score += 20;   // Problematic seller
        else if (seller.ViolationCount >= 3) score += 15;
        else if (seller.ViolationCount >= 1) score += 10;
        
        // 5. CASE TYPE (10 points)
        if (dispute.Type == "INR") score += 10;        // Item not received = urgent
        else if (dispute.Type == "Counterfeit") score += 10;
        
        // 6. ESCALATION (10 points)
        if (dispute.Status == "Escalated") score += 10;
        
        // CLASSIFY
        if (score >= 80) return "Critical";  // ?? Red
        if (score >= 60) return "High";      // ?? Orange
        if (score >= 40) return "Medium";    // ?? Yellow
        return "Low";                         // ? White
    }
}

## 5?? DECISION RULES (POLICY ENGINE)

### **Rule Set 1: Item Not Received (INR)**

```csharp
if (dispute.Type == "INR")
{
    // Check tracking
    if (tracking.Status == "Delivered" && tracking.HasSignature)
    {
        // Strong evidence for seller
        recommendation = "SellerWins";
        reason = "Tracking shows delivered with signature";
    }
    else if (tracking.Status == "Delivered" && !tracking.HasSignature)
    {
        // Check address
        if (tracking.Address == order.ShippingAddress)
        {
            recommendation = "SellerWins"; // Delivered to correct address
        }
        else
        {
            recommendation = "BuyerWins"; // Wrong address
        }
    }
    else if (tracking.Status == "Lost" || tracking.Status == "Returned")
    {
        recommendation = "BuyerWins"; // Clear case
    }
}
```

### **Rule Set 2: Item Not as Described (INAD)**

```csharp
if (dispute.Type == "INAD")
{
    // Analyze evidence
    var listingDescription = order.Product.Description;
    var buyerClaim = dispute.BuyerEvidence;
    
    // Common scenarios
    if (buyerClaim.Contains("different color"))
    {
        // Check product listing photos vs buyer photos
        if (ColorsAreDifferent(listing, buyer))
        {
            recommendation = "BuyerWins";
            action = "Full refund + Return required";
        }
    }
    else if (buyerClaim.Contains("counterfeit") || buyerClaim.Contains("fake"))
    {
        // Check authentication
        if (buyerHasAuthenticationReport)
        {
            recommendation = "BuyerWins";
            action = "Full refund + Seller violation + Flag product";
        }
        else
        {
            recommendation = "NeedMoreInfo";
            action = "Request authentication report from buyer";
        }
    }
    else if (buyerClaim.Contains("damaged") || buyerClaim.Contains("broken"))
    {
        // Check who's at fault
        if (listing.Condition == "New" && photos show damage)
        {
            recommendation = "BuyerWins";
        }
        else if (listing.Condition == "Used" && damage matches description)
        {
            recommendation = "SellerWins";
        }
    }
}
```

### **Rule Set 3: Counterfeit (VeRO)**

```csharp
if (dispute.Type == "Counterfeit")
{
    // High priority always
    dispute.Priority = "Critical";
    
    // Check VeRO program eligibility
    if (dispute.RaisedByBrandOwner)
    {
        // Brand owner has verified rights
        recommendation = "BuyerWins";
        action = "Immediate removal + Seller warning";
        
        // If seller has 3+ counterfeit violations
        if (seller.CounterfeitViolations >= 3)
        {
            action += " + Account suspension";
        }
    }
    else
    {
        // Regular buyer claims counterfeit
        if (buyer.HasAuthenticationCertificate)
        {
            recommendation = "BuyerWins";
        }
        else
        {
            recommendation = "RequireAuthentication";
            action = "Request buyer to get item authenticated";
        }
    }
}
```
